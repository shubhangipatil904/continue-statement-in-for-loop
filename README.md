# continue-statement-in-for-loop
Using System;
Namespace CSharp_Shell
{
    Public class Program 
    {
        Public static void Main()
        {
	For(int x = 1; x <= 12; x++) 
    { 
         If (x == 3) 
           { 
               Continue; 
            } 
            Console.WriteLine(x);
        }   
  }   
  } 
  }
